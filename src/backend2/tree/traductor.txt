void mostrarFactorial (double n2){

	for(double i = n2; i > 0; i = i - 1){
		fact = fact * i;
		str = str + i;
		if (i > 1){
			str = str + " * ";

			str = str + " = ";
}
}
}

print(str, "\n");
}
